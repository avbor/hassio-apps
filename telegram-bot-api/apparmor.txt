#include <tunables/global>

profile telegram-bot-api flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>
  #include <abstractions/openssl>
  #include <abstractions/consoles>

  # Init and shell
  /bin/sh rmix,
  /bin/bash rmix,
  /init rmix,

  # S6-Overlay
  /etc/s6-overlay/** r,
  /etc/s6-overlay/s6-rc.d/** r,
  /usr/lib/s6-overlay/** rmix,
  /bin/** rmix,
  /usr/bin/** rmix,
  /usr/local/bin/** rmix,
  /lib/** rm,
  /usr/lib/** rm,

  # Capabilities
  capability kill,
  capability setuid,
  capability setgid,
  capability chown,
  capability dac_override,

  # Signals
  signal (send) set=(kill,term,int,hup,cont),

  # S6
  /etc/s6-linux-init/** r,
  /run/s6/** rwk,
  /etc/cont-init.d/** rmix,
  /etc/services.d/** rmix,
  /etc/s6-linux-init/** rmix,
  /run/{s6,s6-rc*,service}/** rmix,
  /package/** rmix,
  /command/** rmix,
  /bin/s6-rc rmix,
  /bin/s6-rc-compile rmix,
  /bin/s6-svc rmix,
  /bin/s6-svscan rmix,

  # Network
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # System folders and temp
  /etc/passwd r,
  /etc/group r,
  /etc/nsswitch.conf r,
  /etc/resolv.conf r,
  /etc/hosts r,
  /run/{,**} rwk,
  /tmp/** rwk,
  /dev/tty rw,
  /dev/null rw,
  /dev/urandom r,

  # App Folders
  /data/{,**} rw,
  /media/{,**} rw,

  /usr/local/bin/telegram-bot-api cx -> telegram_server,

  profile telegram_server flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>
    #include <abstractions/openssl>

    network inet stream,
    network inet6 stream,

    signal (receive) peer=telegram-bot-api,

    /data/{,**} rw,
    /media/{,**} rw,
    /usr/local/bin/telegram-bot-api r,
    /lib/** rm,
    /usr/lib/** rm,
    
    /etc/ssl/certs/ca-certificates.crt r,
    /etc/resolv.conf r,
    /etc/hosts r,
    /dev/urandom r,
    /dev/null rw,
  }
}
